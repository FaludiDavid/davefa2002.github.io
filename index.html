import yfinance as yf
import pandas as pd
import plotly.express as px
import plotly.io as pio

# Plotly Colab megjelenГ­tГ©s
pio.renderers.default = "colab"

# KriptГіk Г©s szimbГіlumaik
symbols = {
    "Bitcoin": "BTC-USD",
    "Ethereum": "ETH-USD",
    "Dogecoin": "DOGE-USD"
}

# LetГ¶ltГ©s Г©s egysГ©gesГ­tГ©s
combined = []

for name, ticker in symbols.items():
    df = yf.download(ticker, period="5y", interval="1d")
    df.reset_index(inplace=True)
    df["Crypto"] = name
    df["Date"] = pd.to_datetime(df["Date"])
    df["Price"] = df["Close"]
    combined.append(df[["Date", "Price", "Crypto"]])

# EgyesГ­tett DataFrame
combined = pd.concat(combined, ignore_index=True)

"""                        14 napos szГЎzalГ©kos vГЎltozГЎs szГЎmГ­tГЎsa (rolling average)"""

# DГЎtum szerint rendezГ©s
combined = combined.sort_values(by=["Crypto", "Date"])

# 14 napos vГЎltozГЎs fГјggvГ©ny
def compute_pct_rolling_avg(group):
    group = group.copy()
    group["RollingAvg"] = group["Price"].pct_change(periods=14) * 100
    return group

# SzГЎmГ­tГЎs kГјlГ¶n-kГјlГ¶n kriptГіkra
combined_data_smooth = combined.groupby("Crypto").apply(compute_pct_rolling_avg).reset_index(drop=True)

fig = px.line(
    combined_data_smooth,
    x="Date",
    y="RollingAvg",
    color="Crypto",
    line_group="Crypto",
    hover_data={"Date": True, "Crypto": True, "RollingAvg": ':.2f'},
    title="рџ“‰ 5 Г©ves mozgГіГЎtlag вЂ“ ГЃrfolyamvГЎltozГЎs Г¶sszehasonlГ­tГЎs"
)

fig.update_traces(
    hovertemplate='<b>DГЎtum:</b> %{x}<br>' +
                  '<b>Crypto:</b> %{customdata[1]}<br>' +
                  '<b>14 napos vГЎltozГЎs:</b> %{y:.2f}%<extra></extra>'
)

fig.update_layout(
    xaxis_title="DГЎtum",
    yaxis_title="VГЎltozГЎs (%)",
    hovermode="x unified",
    template="plotly_white"
)

fig.show()

import pandas as pd
import plotly.express as px
import yfinance as yf

# KriptГіk szimbГіlumai
symbols = {
    "Bitcoin": "BTC-USD",
    "Ethereum": "ETH-USD",
    "Dogecoin": "DOGE-USD"
}

# Adatok letГ¶ltГ©se Г©s elЕ‘kГ©szГ­tГ©se
combined = []
for name, ticker in symbols.items():
    df = yf.download(ticker, period="5y", interval="1d")
    df.reset_index(inplace=True)
    df["Crypto"] = name
    df["Date"] = pd.to_datetime(df["Date"])
    df["Price"] = df["Close"]
    df["Volume_K"] = df["Volume"] / 1e3  # Volumen K-ban (ezer USD)
    df["Month"] = df["Date"].dt.month_name()
    df["Year"] = df["Date"].dt.year
    combined.append(df[["Date", "Price", "Volume_K", "Crypto", "Month", "Year"]])

combined_df = pd.concat(combined, ignore_index=True)

# ГЃtlagos havi forgalom kiszГЎmГ­tГЎsa Bitcoin Г©s Ethereum esetГ©n
heatmap_data = combined_df[combined_df["Crypto"].isin(["Bitcoin", "Ethereum"])].copy()
heatmap_data = (
    heatmap_data.groupby(["Crypto", "Year", "Month"], as_index=False)["Volume_K"]
    .mean()
)

# HЕ‘tГ©rkГ©p - Bitcoin Г©s Ethereum
fig1 = px.density_heatmap(
    heatmap_data,
    x="Month",
    y="Year",
    z="Volume_K",
    facet_col="Crypto",
    color_continuous_scale="YlGnBu",
    title="рџ“Љ ГЃtlagos havi forgalom (K USD) вЂ“ Bitcoin vs Ethereum",
    labels={"Volume_K": "Forgalom (K USD)"}
